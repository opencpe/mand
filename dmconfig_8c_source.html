127 uint32_t
128 event_aux_diamRead(int fd, short event, COMMCONTEXT *readCtx,
129  uint8_t *alreadyRead, COMMSTATUS *status)
130 {
131  ssize_t length;
132  uint32_t bufsize;
133  uint32_t len;
134 
135  if (event == EV_TIMEOUT)
136  aux_RET(CONNRESET, RC_ERR_CONNECTION);
137 
138  if (event != EV_READ) /* a number of events can be ignored */
139  aux_RET(INCOMPLETE, RC_OK);
140 
141  /* nothing or not enough read -> read as much as possible */
142 
143  if (!*alreadyRead) {
144  *alreadyRead = 1;
145 
146  bufsize = readCtx->cAlloc * BUFFER_CHUNK_SIZE;
147  do {
148  /* don't reallocate if there's still space */
149  if (readCtx->bytes == bufsize) {
150  readCtx->cAlloc++;
151  bufsize += BUFFER_CHUNK_SIZE;
152 
153  /* allocate a DIAM_REQUEST structure (we're reading into the REQUEST structure) */
154  if (!(readCtx->req = talloc_realloc_size(NULL, readCtx->req,
155  sizeof(DIAM_REQUEST_INFO) + bufsize)))
156  aux_RET(ERROR, RC_ERR_ALLOC);
157 
158  readCtx->buffer = (uint8_t*)readCtx->req +
159  sizeof(DIAM_REQUEST_INFO);
160  }
161 
162  do {
163  if ((length = read(fd, readCtx->buffer + readCtx->bytes,
164  bufsize - readCtx->bytes)) == -1) {
165  debug(": read error: %d (%m)", errno);
166  switch (errno) {
167  case EWOULDBLOCK: /* happens if data to read is multiple of BUFFER_CHUNK_SIZE */
168  length = 0;
169  break;
170  case EINTR:
171  break;
172  case ETIMEDOUT:
173  case ECONNRESET:
174  aux_RET(CONNRESET,
175  RC_ERR_CONNECTION);
176  default:
177  aux_RET(ERROR, RC_ERR_CONNECTION);
178  }
179  }
180  else if (!length)
181  aux_RET(CONNRESET, RC_ERR_CONNECTION);
182  /* if length is nevertheless 0, there was an EWOULDBLOCK */
183  } while (length == -1); /* errno is EINTR */
184 
185  readCtx->bytes += length;
186  debug(": read length: %d, total: %d (from %d)", (int)length, readCtx->bytes, bufsize);
187  } while (length && readCtx->bytes == bufsize);
188  } else if (readCtx->bytes >= 4 &&
189  readCtx->bytes >=
190  (len = diam_packet_length(&readCtx->req->packet))) {
191 
192  debug(": read continue, got total: %d, req: %d", readCtx->bytes, len);
193 
194  /* foremost request was also the last one */
195  if (!(readCtx->bytes - len)) {
196  debug(": remove request");
197  talloc_free(readCtx->req);
198  readCtx->req = NULL;
199  readCtx->cAlloc = readCtx->bytes = 0;
200  aux_RET(NOTHING, RC_OK);
201  }
202 
203  /* remove request */
204  readCtx->bytes -= len;
205  memmove(readCtx->buffer, readCtx->buffer + len, readCtx->bytes);
206  }
207 
208  /* process read data */
209 
210  /* less than a part of one request's header to process */
211  if (readCtx->bytes < 4 ||
212  readCtx->bytes < diam_packet_length(&readCtx->req->packet))
213  aux_RET(INCOMPLETE, RC_OK);
214 
215  postprocessRequest(readCtx); /* one or more requests to process */
216  aux_RET(COMPLETE, RC_OK);
217 }

219 uint32_t
220 event_aux_diamWrite(int fd, short event, COMMCONTEXT *writeCtx,
221  COMMSTATUS *status)
222 {
223  ssize_t length;
224 
225  struct sigaction action, oldaction;
226 
227  if (event == EV_TIMEOUT)
228  aux_RET(CONNRESET, RC_ERR_CONNECTION);
229 
230  if (event != EV_WRITE)
231  aux_RET(INCOMPLETE, RC_OK);
232 
233  if (!writeCtx->buffer) {
234  writeCtx->buffer = (uint8_t *)&writeCtx->req->packet;
235  writeCtx->bytes = writeCtx->req->info.size -
236  sizeof(DIAM_REQUEST_INFO);
237  }
238 
239  memset(&action, 0, sizeof(struct sigaction));
240  action.sa_handler = SIG_IGN;
241  sigaction(SIGPIPE, &action, &oldaction);
242 
243  do {
244  if ((length = write(fd, writeCtx->buffer,
245  writeCtx->bytes)) == -1)
246  switch (errno) {
247  case EAGAIN: /* can only happen when it tries to write the second request in one write event */
248  aux_RET_SIG(NOTHING, RC_OK);
249  case EPIPE:
250  case ECONNRESET:
251  aux_RET_SIG(CONNRESET, RC_ERR_CONNECTION);
252  case EINTR:
253  break;
254  default:
255  aux_RET_SIG(ERROR, RC_ERR_CONNECTION);
256  }
257  else if (!length)
258  aux_RET_SIG(CONNRESET, RC_ERR_CONNECTION);
259  } while (length == -1); /* errno is EINTR */
260 
261  writeCtx->buffer += length;
262  writeCtx->bytes -= length;
263 
264  aux_RET_SIG((writeCtx->bytes ? INCOMPLETE : COMPLETE), RC_OK);
265 }

288 uint32_t
289 dm_register_connect_callback(DMCONTEXT *dmCtx, int type,
290  DMCONFIG_CONNECT_CALLBACK callback, void *userdata)
291 {
292  CONNEVENTCTX *ctx;
293  struct timeval timeout;
294 
295  int fd = dm_context_get_socket(dmCtx);
296  int flags;
297 
298  struct sockaddr_un sockaddr_un;
299  struct sockaddr_in sockaddr_in;
300 
301  struct sockaddr *sockaddr;
302  socklen_t sockaddr_len;
303 
304  if ((flags = fcntl(fd, F_GETFL)) == -1 ||
305  fcntl(fd, F_SETFL, flags | O_NONBLOCK))
306  return RC_ERR_CONNECTION;
307 
308  if (type == AF_UNIX) {
309  memset(&sockaddr_un, 0, sizeof(sockaddr_un));
310 
311  sockaddr_un.sun_family = AF_UNIX;
312  strncpy(sockaddr_un.sun_path + 1, SERVER_LOCAL,
313  sizeof(sockaddr_un.sun_path) - 1);
314 
315  sockaddr = (struct sockaddr*)&sockaddr_un;
316  sockaddr_len = sizeof(sockaddr_un);
317  } else { /* AF_INET */
318  memset(&sockaddr_in, 0, sizeof(sockaddr_in));
319 
320  sockaddr_in.sin_family = AF_INET;
321  sockaddr_in.sin_port = htons(SERVER_PORT);
322  sockaddr_in.sin_addr.s_addr = htonl(SERVER_IP);
323 
324  sockaddr = (struct sockaddr*)&sockaddr_in;
325  sockaddr_len = sizeof(sockaddr_in);
326  }
327 
328  while (connect(fd, sockaddr, sockaddr_len) == -1)
329  if (errno == EINPROGRESS)
330  break;
331  else if (errno == EAGAIN)
332  continue;
333  else
334  return RC_ERR_CONNECTION;
335 
336  if (!(ctx = talloc(NULL, CONNEVENTCTX)))
337  return RC_ERR_ALLOC;
338 
339  ctx->callback = callback;
340  ctx->user_data = userdata;
341  ctx->dmCtx = dmCtx;
342 
343  event_set(&ctx->event, fd, EV_WRITE, connectEvent, ctx);
344  event_base_set(dm_context_get_event_base(dmCtx), &ctx->event);
345 
346  timeout.tv_sec = TIMEOUT_WRITE_REQUESTS; /* if the connect was already successful, there should be no delay */
347  timeout.tv_usec = 0;
348 
349  if (event_add(&ctx->event, &timeout)) {
350  talloc_free(ctx);
351  return RC_ERR_ALLOC;
352  }
353 
354  return RC_OK;
355 }

357 uint32_t
358 dm_generic_register_request(DMCONTEXT *dmCtx, uint32_t code, DIAM_AVPGRP *grp,
359  DMCONFIG_CALLBACK callback, void *callback_ud)
360 {
361  DIAM_AVPGRP *completegrp;
362 
363  REQUESTINFO *new, *cur;
364 
365  struct timeval timeout;
366 
367  switch (hopid) { /* one never knows... */
368  case 0: srand((unsigned int)time(NULL));
369  case MAX_INT: hopid = endid = (float)rand()/RAND_MAX * (MAX_INT-1) + 1;
370  break;
371  default: hopid = ++endid;
372  }
373 
374  if (!dmCtx->requestlist_head) {
375  if (!(dmCtx->requestlist_head = talloc(NULL, REQUESTINFO)))
376  return RC_ERR_ALLOC;
377  memset(dmCtx->requestlist_head, 0, sizeof(REQUESTINFO));
378  }
379  if (!(new = talloc(dmCtx->requestlist_head, REQUESTINFO))) {
380  talloc_free(dmCtx->requestlist_head);
381  return RC_ERR_ALLOC;
382  }
383 
384  if (!(new->request = new_diam_request(new, code, CMD_FLAG_REQUEST,
385  APP_ID, hopid, endid))) {
386  talloc_free(dmCtx->requestlist_head);
387  return RC_ERR_ALLOC;
388  }
389 
390  if (!(completegrp = new_diam_avpgrp(new->request)) ||
391  diam_avpgrp_add_uint32(new->request, &completegrp, AVP_SESSIONID, 0,
392  VP_TRAVELPING, dmCtx->sessionid) ||
393  (grp && diam_avpgrp_add_avpgrp(new->request, &completegrp,
394  AVP_CONTAINER, 0, VP_TRAVELPING,
395  grp)) ||
396  build_diam_request(new, &new->request, completegrp)) {
397  talloc_free(dmCtx->requestlist_head);
398  return RC_ERR_ALLOC;
399  }
400  talloc_free(completegrp);
401 
402 #ifdef LIBDMCONFIG_DEBUG
403  if (dmconfig_debug_level) {
404  fprintf(stderr, "Send request:\n");
405  dump_diam_packet(new->request);
406  }
407 #endif
408 
409  new->callback = callback;
410  new->user_data = callback_ud;
411  new->dmCtx = dmCtx;
412  new->status = REQUEST_SHALL_WRITE;
413  new->code = code;
414  new->hopid = hopid;
415  new->next = NULL;
416 
417  for (cur = dmCtx->requestlist_head; cur->next; cur = cur->next);
418  cur->next = new;
419 
420  if (!event_pending(&dmCtx->writeCtx.event, EV_WRITE | EV_PERSIST, NULL)) {
421  event_set(&dmCtx->writeCtx.event, dmCtx->socket,
422  EV_WRITE | EV_PERSIST, writeEvent, dmCtx);
423  event_base_set(dm_context_get_event_base(dmCtx), &dmCtx->writeCtx.event);
424 
425  timeout.tv_sec = TIMEOUT_WRITE_REQUESTS;
426  timeout.tv_usec = 0;
427 
428  if (event_add(&dmCtx->writeCtx.event, &timeout)) {
429  talloc_free(dmCtx->requestlist_head);
430  return RC_ERR_ALLOC;
431  }
432  }
433 
434  return RC_OK;
435 }

793 static void
794 writeEvent(int fd, short event, void *arg)
795 {
796  DMCONTEXT *dmCtx = arg;
797  COMMCONTEXT *ctx = &dmCtx->writeCtx;
798 
799  COMMSTATUS status;
800  uint32_t rc;
801 
802  struct timeval timeout;
803 
804  do {
805  if (!ctx->buffer) {
806  for (ctx->cur_request = dmCtx->requestlist_head->next;
807  ctx->cur_request &&
808  ctx->cur_request->status != REQUEST_SHALL_WRITE;
809  ctx->cur_request = ctx->cur_request->next);
810  if (!ctx->cur_request) { /* all requests written */
811  event_del(&ctx->event);
812  return;
813  }
814  ctx->cur_request->status = REQUEST_WRITING;
815  ctx->req = ctx->cur_request->request;
816  }
817 
818  rc = event_aux_diamWrite(fd, event, ctx, &status);
819  switch (status) {
820  case COMPLETE:
821  talloc_free(ctx->cur_request->request);
822  ctx->buffer = NULL;
823 
824  ctx->cur_request->status = REQUEST_SHALL_READ;
825 
826  if (!event_pending(&dmCtx->readCtx.event,
827  EV_READ | EV_PERSIST, NULL)) {
828  event_set(&dmCtx->readCtx.event, dmCtx->socket,
829  EV_READ | EV_PERSIST, readEvent, dmCtx);
830  event_base_set(dm_context_get_event_base(dmCtx),
831  &dmCtx->readCtx.event);
832 
833  timeout.tv_sec = TIMEOUT_READ_REQUESTS;
834  timeout.tv_usec = 0;
835 
836  if (event_add(&dmCtx->readCtx.event, &timeout)) {
837  rc = RC_ERR_ALLOC;
838  break;
839  }
840  }
841 
842  timeout.tv_sec = TIMEOUT_WRITE_REQUESTS;
843  timeout.tv_usec = 0;
844 
845  if (event_add(&ctx->event, &timeout)) /* increase writeEvent's timeout */
846  rc = RC_ERR_ALLOC;
847 
848  break;
849  case INCOMPLETE:
850  timeout.tv_sec = TIMEOUT_CHUNKS;
851  timeout.tv_usec = 0;
852 
853  if (event_add(&ctx->event, &timeout)) { /* reduce writeEvent's timeout */
854  rc = RC_ERR_ALLOC;
855  break;
856  }
857  case NOTHING:
858  return;
859  default: /* CONNRESET or ERROR */
860  break;
861  }
862  } while (!rc);
863 
864  CALLBACK(ctx->cur_request, DMCONFIG_ERROR_WRITING,
865  ctx->cur_request->dmCtx, ctx->cur_request->user_data, rc, NULL);
866 
867  L_FOREACH(REQUESTINFO, cur, dmCtx->requestlist_head)
868  if (cur->status != REQUEST_SHALL_READ)
869  talloc_free(cur->request);
870  talloc_free(dmCtx->requestlist_head);
871  dmCtx->requestlist_head = NULL;
872 
873  ctx->cur_request = NULL;
874  ctx->req = NULL;
875  ctx->buffer = NULL;
876  event_del(&ctx->event);
877 
878  talloc_free(dmCtx->readCtx.req);
879  dmCtx->readCtx.req = NULL;
880 
881  if (event_pending(&dmCtx->readCtx.event, EV_READ | EV_PERSIST, NULL))
882  event_del(&dmCtx->readCtx.event);
883 }

885 static void
886 readEvent(int fd, short event, void *arg)
887 {
888  DMCONTEXT *dmCtx = arg;
889  COMMCONTEXT *ctx = &dmCtx->readCtx;
890 
891  uint32_t avpcode;
892  uint8_t flags;
893  uint32_t vendor_id;
894  void *data;
895  size_t len;
896 
897  uint8_t alreadyRead = 0;
898 
899  struct timeval timeout;
900 
901  for (;;) {
902  uint32_t hopid;
903  REQUESTINFO *cur, *reqEl;
904 
905  COMMSTATUS status;
906  uint32_t rc;
907 
908  rc = event_aux_diamRead(fd, event, ctx, &alreadyRead, &status);
909  switch (status) {
910  case INCOMPLETE:
911  timeout.tv_sec = TIMEOUT_CHUNKS;
912  timeout.tv_usec = 0;
913 
914  if (event_add(&ctx->event, &timeout)) /* reduce readEvent's timeout */
915  goto abort;
916  case NOTHING:
917  return;
918  case COMPLETE:
919  break;
920  default: /* CONNRESET or ERROR */
921  goto abort;
922  }
923 
924 #ifdef LIBDMCONFIG_DEBUG
925  if (dmconfig_debug_level) {
926  fprintf(stderr, "Recieved %s:\n",
927  diam_packet_flags(&ctx->req->packet) &
928  CMD_FLAG_REQUEST ? "request" : "answer");
929  dump_diam_packet(ctx->req);
930  diam_request_reset_avp(ctx->req);
931  }
932 #endif
933 
934  /* server request */
935  if (diam_packet_flags(&ctx->req->packet) & CMD_FLAG_REQUEST) {
936  switch (diam_packet_code(&ctx->req->packet)) {
937 
938  default:
939  goto abort;
940  }
941  } else {
942  if (!dmCtx->requestlist_head || !dmCtx->requestlist_head->next)
943  goto abort;
944 
945  hopid = diam_hop2hop_id(&ctx->req->packet);
946 
947  for (cur = dmCtx->requestlist_head;
948  cur->next && cur->next->hopid != hopid;
949  cur = cur->next);
950  if (!cur->next || cur->next->status != REQUEST_SHALL_READ)
951  goto abort;
952  reqEl = cur->next;
953  cur->next = reqEl->next;
954 
955  if (diam_request_get_avp(ctx->req, &avpcode, &flags,
956  &vendor_id, &data, &len) ||
957  avpcode != AVP_RC || len != sizeof(uint32_t)) {
958  CALLBACK(reqEl, DMCONFIG_ERROR_READING, reqEl->dmCtx, reqEl->user_data, RC_ERR_MISC, NULL);
959  goto cleanup;
960  }
961 
962  if ((rc = diam_get_uint32_avp(data))) {
963  CALLBACK(reqEl, DMCONFIG_ANSWER_READY, reqEl->dmCtx, reqEl->user_data, rc, NULL);
964 
965  /* clean up callback structures if necessary, so the read event is deleted */
966  switch (reqEl->code) {
967  case CMD_SUBSCRIBE_NOTIFY:
968  memset(&dmCtx->callbacks.active_notification, 0,
969  sizeof(ACTIVE_NOTIFY_INFO));
970  break;
971  }
972 
973  goto cleanup;
974  } else /* RC_OK */ {
975  switch (reqEl->code) {
976  case CMD_UNSUBSCRIBE_NOTIFY:
977  memset(&dmCtx->callbacks.active_notification, 0,
978  sizeof(ACTIVE_NOTIFY_INFO));
979  break;
980 
981  case CMD_ENDSESSION: /*
982  * allows the implicit abortion (deletion of read event -> event loop returns) of
983  * asynchronous processes (ping, active notify, etc)
984  */
985  memset(&dmCtx->callbacks, 0,
986  sizeof(struct _dmContext_callbacks));
987  break;
988  }
989  }
990 
991  if (diam_request_get_avp(ctx->req, &avpcode, &flags, &vendor_id, &data, &len)) {
992  CALLBACK(reqEl, DMCONFIG_ANSWER_READY, reqEl->dmCtx, reqEl->user_data, RC_OK, NULL);
993  } else if (avpcode == AVP_CONTAINER) {
994  DIAM_AVPGRP *answer;
995 
996  if ((answer = diam_decode_avpgrp(ctx->req, data, len))) {
997  CALLBACK(reqEl, DMCONFIG_ANSWER_READY, reqEl->dmCtx, reqEl->user_data, RC_OK, answer);
998  } else {
999  CALLBACK(reqEl, DMCONFIG_ERROR_READING, reqEl->dmCtx, reqEl->user_data, RC_ERR_ALLOC, NULL);
1000  }
1001  } else {
1002  CALLBACK(reqEl, DMCONFIG_ERROR_READING, reqEl->dmCtx, reqEl->user_data, RC_ERR_MISC, NULL);
1003  }
1004 
1005 cleanup:
1006 
1007  if (dmCtx->requestlist_head->next)
1008  talloc_free(reqEl);
1009  else {
1010  talloc_free(dmCtx->requestlist_head);
1011  dmCtx->requestlist_head = NULL;
1012  }
1013  }
1014 
1015  for (cur = dmCtx->requestlist_head;
1016  cur && cur->status != REQUEST_SHALL_READ;
1017  cur = cur->next);
1018  if (!cur && /* nothing more to read (at least not expected) */
1019  !dmCtx->callbacks.active_notification.callback) { /* FIXME: a reference counter would be cleaner */
1020  event_del(&ctx->event);
1021  /* if there's nothing more to read, the fields are reset and NOTHING is returned */
1022  event_aux_diamRead(fd, EV_READ, ctx,
1023  &alreadyRead, &status);
1024  if (status != NOTHING)
1025  goto abort;
1026 
1027  return;
1028  }
1029 
1030  timeout.tv_sec = TIMEOUT_READ_REQUESTS;
1031  timeout.tv_usec = 0;
1032 
1033  if (event_add(&ctx->event, &timeout)) /* increase readEvent's timeout */
1034  break;
1035  }
1036 
1037 abort:
1038 
1039  event_del(&ctx->event);
1040  talloc_free(ctx->req);
1041  ctx->req = NULL;
1042  ctx->cAlloc = ctx->bytes = 0;
1043  memset(&dmCtx->callbacks, 0, sizeof(struct _dmContext_callbacks));
1044 }

1242  /* converts an arbitrary typed AVP data to an ASCII string */
1243 uint32_t
1244 dm_decode_unknown_as_string(uint32_t type, void *data, size_t len, char **val)
1245 {

1248  struct in_addr in;
1249  struct in6_addr in6;
1250  } addr;
1251  char *dum;
1252 
1253  switch (type) {
1254  case AVP_BOOL:
1255  return (*val = strdup(diam_get_uint8_avp(data) ? "1" : "0"))
1256  ? RC_OK : RC_ERR_ALLOC;
1257  case AVP_ENUMID:
1258  case AVP_INT32:

1260  ? RC_ERR_ALLOC : RC_OK;
1261  case AVP_COUNTER:

1264  ? RC_ERR_ALLOC : RC_OK;

1266  case AVP_RELTICKS:
1267  case AVP_INT64:
1268  return asprintf(val, "%" PRIi64, diam_get_int64_avp(data)) == -1
1269  ? RC_ERR_ALLOC : RC_OK;
1270  case AVP_UINT64:
1271  return asprintf(val, "%" PRIu64, diam_get_uint64_avp(data)) == -1
1272  ? RC_ERR_ALLOC : RC_OK;
1273  case AVP_ENUM:
1274  case AVP_PATH:
1275  case AVP_STRING:
1276  return (*val = strndup(data, len)) ? RC_OK : RC_ERR_ALLOC;
1277  case AVP_BINARY: {
1278  *val = malloc(((len + 3) * 4) / 3);

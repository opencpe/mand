248 Lua_decode_list(lua_State *L, DIAM_AVPGRP *grp, int *nodes)
249 {
250  uint32_t code;
251  uint8_t flags;
252  uint32_t vendor_id;
253  void *data;
254  size_t len;
255 
256  uint32_t rc;
257  int i;
258 
259  lua_newtable(L);
260 
261  for (i = 1;
262  !diam_avpgrp_get_avp(grp, &code, &flags, &vendor_id, &data, &len);
263  i++) {
264  DIAM_AVPGRP *node_container;
265  uint32_t type;
266 
267  if (code != AVP_CONTAINER || !len)
268  return RC_ERR_MISC;
269  lua_pushinteger(L, i);
270 
271  if (!(node_container = diam_decode_avpgrp(NULL, data, len)))
272  return RC_ERR_ALLOC;
273  lua_newtable(L);
274 
275  if (diam_avpgrp_get_avp(node_container, &code, &flags, &vendor_id,
276  &data, &len) ||
277  code != AVP_NODE_NAME || !len) {
278  dm_grp_free(node_container);
279  return RC_ERR_MISC;
280  }
281  lua_pushlstring(L, data, len);
282  lua_setfield(L, -2, "name");
283 
284  if (diam_avpgrp_get_avp(node_container, &code, &flags, &vendor_id,
285  &data, &len) ||
286  code != AVP_NODE_TYPE || len != sizeof(uint32_t)) {
287  dm_grp_free(node_container);
288  return RC_ERR_MISC;
289  }
290  type = diam_get_uint32_avp(data);
291  lua_pushinteger(L, type);
292  lua_setfield(L, -2, "type");
293 
294  switch (type) {
295  case NODE_PARAMETER:
296  if (diam_avpgrp_get_avp(node_container, &code, &flags,
297  &vendor_id, &data, &len)) {
298  dm_grp_free(node_container);
299  return RC_ERR_MISC;
300  }
301 
302  if (code == AVP_NODE_DATATYPE) {
303  if (len != sizeof(uint32_t)) {
304  dm_grp_free(node_container);
305  return RC_ERR_MISC;
306  }
307 
308  lua_pushinteger(L, diam_get_uint32_avp(data));
309  lua_setfield(L, -2, "datatype");
310  } else {
311  lua_pushinteger(L, code);
312  lua_setfield(L, -2, "datatype");
313 
314  if ((rc = Lua_decode_unknown(L, code, data, len))) {
315  dm_grp_free(node_container);
316  return rc;
317  }
318  lua_setfield(L, -2, "value");
319  }
320 
321  break;
322 
323  case NODE_TABLE:
324  case NODE_OBJECT: {
325  DIAM_AVPGRP *child_grp;
326  int child_nodes;
327 
328  if (type == NODE_TABLE) {
329  if (diam_avpgrp_get_avp(node_container, &code, &flags,
330  &vendor_id, &data, &len))
331  break;
332  } else {
333  if (diam_avpgrp_get_avp(node_container, &code, &flags,
334  &vendor_id, &data, &len)) {
335  dm_grp_free(node_container);
336  return RC_ERR_MISC;
337  }
338 
339  if (code == AVP_NODE_SIZE) {
340  if (len != sizeof(uint32_t)) {
341  dm_grp_free(node_container);
342  return RC_ERR_MISC;
343  }
344 
345  lua_pushinteger(L, diam_get_uint32_avp(data));
346  lua_setfield(L, -2, "size");
347 
348  break;
349  }
350  }
351 
352  if (code != AVP_CONTAINER) {
353  dm_grp_free(node_container);
354  return RC_ERR_MISC;
355  }
356 
357  if (!(child_grp = diam_decode_avpgrp(node_container, data, len))) {
358  dm_grp_free(node_container);
359  return RC_ERR_ALLOC;
360  }
361 
362  if ((rc = Lua_decode_list(L, child_grp, &child_nodes))) {
363  dm_grp_free(node_container);
364  return rc;
365  }
366  lua_setfield(L, -2, "children");
367 
368  lua_pushinteger(L, child_nodes);
369  lua_setfield(L, -2, "size");
370  }
371  }
372 
373  lua_settable(L, -3);
374  dm_grp_free(node_container);
375  }
376 
377  if (nodes)
378  *nodes = i - 1;
379 
380  return RC_OK;
381 }

476 static void
477 generic_Lua_callback(DMCONFIG_EVENT event, DMCONTEXT *dmCtx, void *user_data,
478  uint32_t answer_rc, DIAM_AVPGRP *answer_grp)
479 {
480  LUA_CALLBACK *cb = user_data;
481  lua_State *L = cb->L;
482 
483  int top;
484 
485  lua_rawgeti(L, LUA_REGISTRYINDEX, cb->callback_ref);
486  top = lua_gettop(L);
487  lua_pushnumber(L, event);
488  lua_rawgeti(L, LUA_REGISTRYINDEX, cb->context_ref);
489  lua_rawgeti(L, LUA_REGISTRYINDEX, cb->user_data_ref);
490  lua_pushnumber(L, answer_rc);
491 
492  if (event == DMCONFIG_ANSWER_READY && !answer_rc)
493  switch (cb->code) {
494  case CMD_STARTSESSION:
495  if (dm_decode_start_session(dmCtx, answer_grp))
496  L_ERROR(L_ALLOC);
497  break;
498  case CMD_ENDSESSION:
499  dm_context_set_sessionid(dmCtx, 0);
500  break;
501  case CMD_SESSIONINFO:
502  if (Lua_decode_get_session_info(L, answer_grp))
503  L_ERROR(L_MISC);
504  break;
505  case CMD_CFGSESSIONINFO: {
506  uint32_t sessionid, flags;
507  struct timeval timeout;
508 
509  if (dm_decode_get_cfg_session_info(answer_grp, &sessionid, &flags, &timeout))
510  L_ERROR(L_ALLOC);
511 
512  Lua_decode_get_cfg_session_info(L, sessionid, flags, timeout);
513  break;
514  }
515  case CMD_DB_FINDINSTANCE:
516  case CMD_DB_ADDINSTANCE: {
517  uint16_t instance;
518 
519  if (dm_decode_uint16(answer_grp, &instance))
520  L_ERROR(L_ALLOC);
521  lua_pushnumber(L, instance);
522  break;
523  }
524  case CMD_DB_DUMP: {
525  char *data;
526 
527  if (dm_decode_string(answer_grp, &data))
528  L_ERROR(L_ALLOC);
529  lua_pushstring(L, data);
530  free(data);
531  break;
532  }
533  case CMD_DB_GET:
534  if (Lua_decode_get(L, answer_grp))
535  L_ERROR(L_ALLOC);
536  break;
537  case CMD_DB_LIST:
538  if (Lua_decode_list(L, answer_grp, NULL))
539  L_ERROR(L_ALLOC);
540  break;
541  case CMD_DB_RETRIEVE_ENUMS:
542  if (Lua_decode_retrieve_enums(L, answer_grp))
543  L_ERROR(L_ALLOC);
544  break;
545 
546  case CMD_GET_PASSIVE_NOTIFICATIONS:
547  if (Lua_decode_notifications(L, answer_grp))
548  L_ERROR(L_ALLOC);
549  break;
550  }
551 
552  lua_call(L, lua_gettop(L) - top, 0);
553 
554  luaL_unref(L, LUA_REGISTRYINDEX, cb->context_ref);
555  luaL_unref(L, LUA_REGISTRYINDEX, cb->callback_ref);
556  luaL_unref(L, LUA_REGISTRYINDEX, cb->user_data_ref);
557 
558  cb->callbackDone = 1; /* req. will be removed from the request list */
559 }

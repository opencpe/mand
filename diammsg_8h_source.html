374 diam_avpgrp_add_time(void *ctx, DIAM_AVPGRP **avpgrp,
375  uint32_t code, uint8_t flags, uint32_t vendor_id,
376  time_t data) {
377  uint32_t dtime;
378

381 
382  dtime = htonl((uint32_t)data + TIME_SINCE_EPOCH);

386  /* accept only timevals with both fields <= 32 bits (standardization) */
387 static inline int
388 diam_avpgrp_add_timeval(void *ctx, DIAM_AVPGRP **avpgrp,
389  uint32_t code, uint8_t flags, uint32_t vendor_id,
390  struct timeval value) {
391  DIAM_TIMEVAL tvalue;
392 
393  if(value.tv_sec > (int32_t)0x7FFFFFFF || value.tv_usec >= 1000000)
394  return 1;
395 
396  tvalue.tv_sec = htonl((uint32_t)value.tv_sec);
397  tvalue.tv_usec = htonl((uint32_t)value.tv_usec);
398 
399  return diam_avpgrp_add_raw(ctx, avpgrp, code, flags, vendor_id, &tvalue, sizeof(DIAM_TIMEVAL));
400 }

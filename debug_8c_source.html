152 static int comp_code(const void *m1, const void *m2)
153 {
154  struct code2str *a = (struct code2str *) m1;
155  struct code2str *b = (struct code2str *) m2;
156 
157  if (a->code > b->code)
158  return 1;
159  else if (a->code < b->code)
160  return -1;
161  else
162  return 0;
163 }
